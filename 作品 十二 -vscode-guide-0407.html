import { App } from './components/App';
import { Router } from './router';
import { Logger } from './utils/logger';
import './styles/global.css';
 
// ── 型別定義 ──────────────────────────────
interface AppConfig {
title: string;
version: string;
routes: Route[]; // 🔍 hover to see type
debug?: boolean;
}
 
// ── 設定常數 ──────────────────────────────
const config: AppConfig = {
title: 'My VS Code Project',
version: '1.0.0',
debug: true,
routes: loadRoutes(),
};
 
// ── 初始化入口 ────────────────────────────
async function bootstrap(cfg: AppConfig): Promise<void> {
const log = new Logger(cfg.debug);
const router = new Router(cfg.routes);
const app = new App({ router, title: cfg.title });
 
log.info('🚀 Starting', cfg.title);
await app.init();
app.mount(document.getElementById('root')!);
log.success('✅ Ready');
}
 
bootstrap(config).catch(err => {
console.error('❌ Fatal:', err);
process.exit(1);
});
